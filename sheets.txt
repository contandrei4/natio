function doGet(e) {
  try {
    // Verificăm dacă primim date de scouting sau cerere de scanare
    if (e.parameter.scor !== undefined) {
      return saveScoutingData(e);
    } else if (e.parameter.team) {
      return scanTeamData(e.parameter.team);
    }
    return ContentService.createTextOutput("Eroare: Lipsesc parametrii.").setMimeType(ContentService.MimeType.TEXT);
  } catch (err) {
    return ContentService.createTextOutput("Eroare Script: " + err.toString()).setMimeType(ContentService.MimeType.TEXT);
  }
}

function saveScoutingData(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var rawSheet = ss.getSheetByName("RawData");
  
  // Creăm sheet-ul dacă nu există
  if (!rawSheet) {
    rawSheet = ss.insertSheet("RawData");
    rawSheet.appendRow(["Echipa", "R ap", "R dep", "A ap", "A dep", "Scor", "RP", "DATA", "Meci: Auto"]);
  }
  
  var teamId = e.parameter.team ? e.parameter.team.toString().trim() : "0";
  var pos = e.parameter.position || "R ap"; 
  var mNum = e.parameter.matchNum || "?";
  var autoData = e.parameter.autoData || "0(0)";
  var teleScor = e.parameter.scor || "0";
  var rp = e.parameter.rp || "0";
  
  // Determinăm alianța din prima literă a poziției (R sau A)
  var allianceLetter = pos.charAt(0); 
  var matchAutoEntry = allianceLetter + mNum + ": " + autoData;

  var data = rawSheet.getDataRange().getValues();
  var rowIndex = -1;

  // Căutăm dacă echipa există deja
  for (var i = 0; i < data.length; i++) {
    if (data[i][0].toString().replace(/'/g, "").trim() === teamId) {
      rowIndex = i + 1;
      break;
    }
  }

  var colMap = { "R ap": 2, "R dep": 3, "A ap": 4, "A dep": 5 };
  var targetCol = colMap[pos] || 2; 

  if (rowIndex !== -1) {
    // Actualizăm rândul existent prin concatenare
    var oldAuto = rawSheet.getRange(rowIndex, targetCol).getValue().toString();
    var oldTele = rawSheet.getRange(rowIndex, 6).getValue().toString(); 
    var oldRP = rawSheet.getRange(rowIndex, 7).getValue().toString(); 
    var oldMatchInfo = rawSheet.getRange(rowIndex, 9).getValue().toString();

    var newAuto = oldAuto ? oldAuto + ", " + autoData : autoData;
    var newTele = oldTele ? oldTele + ", " + teleScor : teleScor;
    var newRP = oldRP ? oldRP + ", " + rp : rp;
    var newMatchInfo = oldMatchInfo ? oldMatchInfo + ", " + matchAutoEntry : matchAutoEntry;

    rawSheet.getRange(rowIndex, targetCol).setValue("'" + newAuto.replace(/'/g, ""));
    rawSheet.getRange(rowIndex, 6).setValue("'" + newTele.replace(/'/g, ""));
    rawSheet.getRange(rowIndex, 7).setValue("'" + newRP.replace(/'/g, ""));
    rawSheet.getRange(rowIndex, 8).setValue(new Date());
    rawSheet.getRange(rowIndex, 9).setValue("'" + newMatchInfo.replace(/'/g, ""));
    
    return ContentService.createTextOutput("SUCCESS_UPDATED");
  } else {
    // Adăugăm un rând nou
    var newRow = ["'" + teamId, "", "", "", "", "'" + teleScor, "'" + rp, new Date(), "'" + matchAutoEntry];
    newRow[targetCol - 1] = "'" + autoData;
    rawSheet.appendRow(newRow);
    return ContentService.createTextOutput("SUCCESS_ADDED");
  }
}

function scanTeamData(teamToFind) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var rawSheet = ss.getSheetByName("RawData");
  if (!rawSheet) return ContentService.createTextOutput(JSON.stringify({error: true})).setMimeType(ContentService.MimeType.JSON);
  
  var rawVals = rawSheet.getDataRange().getValues();
  var match = rawVals.find(row => row[0].toString().replace(/'/g, "").trim() === teamToFind.toString().trim());
  
  if (!match) return ContentService.createTextOutput(JSON.stringify({error: true})).setMimeType(ContentService.MimeType.JSON);

  const parseList = (val) => {
    if (!val) return [];
    return val.toString().replace(/'/g, "").split(/[\s,]+/).map(n => {
      let num = parseFloat(n.split('(')[0]); 
      return isNaN(num) ? null : num;
    }).filter(n => n !== null);
  };

  const getAvg = (arr) => arr.length > 0 ? (arr.reduce((a, b) => a + b, 0) / arr.length).toFixed(1) : "0.0";
  const getMax = (arr) => arr.length > 0 ? Math.max(...arr) : 0;

  var autoAp = parseList(match[1]).concat(parseList(match[3])); 
  var autoDep = parseList(match[2]).concat(parseList(match[4])); 
  var scores = parseList(match[5]); 
  var rps = parseList(match[6]);

  return ContentService.createTextOutput(JSON.stringify({
    avgGeneral: getAvg(scores),
    maxAutoAp: getMax(autoAp),
    maxAutoDep: getMax(autoDep),
    chartData: scores.slice(-10)
  })).setMimeType(ContentService.MimeType.JSON);
}
