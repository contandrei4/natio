function doGet(e) {
  try {
    const team = e.parameter.team;
    const intel = e.parameter.intel;

    if (intel === "true") {
      const res = scanTeamData(team);
      return ContentService.createTextOutput(JSON.stringify(res)).setMimeType(ContentService.MimeType.JSON);
    }

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheetByName("RawData");
    const data = sheet.getDataRange().getValues();
    const pos = e.parameter.position; 
    const autoData = e.parameter.autoData; 
    const scorTele = e.parameter.scor;
    const rp = e.parameter.rp;
    const searchId = String(team).trim();

    let rowIndex = -1;
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]).split('.')[0].trim() === searchId) {
        rowIndex = i + 1; 
        break;
      }
    }

    if (rowIndex !== -1) {
      updateCellWithComma(sheet, rowIndex, 6, scorTele);
      updateCellWithComma(sheet, rowIndex, 7, rp);
      updateCellWithComma(sheet, rowIndex, 9, pos + " " + autoData);
      let colIdx = (pos === "R ap") ? 2 : (pos === "R dep") ? 3 : (pos === "A ap") ? 4 : 5;
      updateCellWithComma(sheet, rowIndex, colIdx, autoData);
      sheet.getRange(rowIndex, 8).setValue(new Date());
    } else {
      const lastRow = sheet.getLastRow() + 1;
      sheet.getRange(lastRow, 1).setValue(team).setNumberFormat("@");
      let colIdx = (pos === "R ap") ? 2 : (pos === "R dep") ? 3 : (pos === "A ap") ? 4 : 5;
      sheet.getRange(lastRow, colIdx).setValue(autoData).setNumberFormat("@");
      sheet.getRange(lastRow, 6).setValue(scorTele).setNumberFormat("@");
      sheet.getRange(lastRow, 7).setValue(rp).setNumberFormat("@");
      sheet.getRange(lastRow, 8).setValue(new Date());
      sheet.getRange(lastRow, 9).setValue(pos + " " + autoData).setNumberFormat("@");
    }
    return ContentService.createTextOutput("SUCCESS").setMimeType(ContentService.MimeType.TEXT);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({error: err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}

function updateCellWithComma(sheet, row, col, newValue) {
  let range = sheet.getRange(row, col);
  range.setNumberFormat("@"); 
  let oldValue = String(range.getValue() || "").trim();
  let updatedValue = oldValue ? oldValue + ", " + newValue : newValue;
  range.setValue(updatedValue);
}

function scanTeamData(teamId) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const rawSheet = ss.getSheetByName("RawData");
  const analyticsSheet = ss.getSheetByName("Analytics");
  const searchId = String(teamId).trim();
  const data = rawSheet.getDataRange().getValues();
  let targetRow = null;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]).split('.')[0].trim() === searchId) { targetRow = data[i]; break; }
  }
  if (!targetRow) return { error: "Echipa " + searchId + " nu are date." };
  const scoresArray = String(targetRow[5]).split(",").map(s => Number(s.trim())).filter(n => !isNaN(n));
  const autoEntries = String(targetRow[8]).split(",");
  const autoPoints = autoEntries.map(e => {
    let m = e.match(/\d+/);
    return m ? parseInt(m[0]) : 0;
  });
  let strategy = "Informație indisponibilă";
  if (analyticsSheet) {
    const aData = analyticsSheet.getDataRange().getValues();
    for (let j = 1; j < aData.length; j++) {
      if (String(aData[j][0]).split('.')[0].trim() === searchId) { strategy = aData[j][3] || "Fara text"; break; }
    }
  }
  return {
    avgGeneral: (scoresArray.reduce((a,b)=>a+b,0)/Math.max(scoresArray.length, 1)).toFixed(1),
    avgLast3: (scoresArray.slice(-3).reduce((a,b)=>a+b,0)/Math.min(scoresArray.length, 3)).toFixed(1),
    strategy: strategy, chartDataTele: scoresArray, chartDataAuto: autoPoints
  };
}
